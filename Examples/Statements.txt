{STATEMENT NAME NON_NULL}
	\\#If the current word from the reversed 'splitter' array is not equal to null
	{OP1 {VAR {VAR INDEX}_splitter}}
	{CMP ne}
	{OP2 (NULL)}
		{STRINGCHECK}
		{ELSE}
		\\#do nothing
		(NULL)
{STATEMENT END}

{STATEMENT NAME STRINGCHECK}
	\\#If the current word is tring
	{OP1 {VAR {VAR INDEX}_splitter}}
	{CMP eq}
	{OP2 string}
		\\#Break the current word into a char array, save it as a STRINGVARCHAR array
		{VAR STRINGVAR={VAR {VAR INDEX}_splitter}}
		{MANIP TCA STRINGVAR,STRINGVARCHAR}

		\\#Reverse the char array
		{MANIP REV STRINGVARCHAR}
		
		\\#Join with nothing and prepend with a comma
		{VAR OUTPUT={MANIP APP {VAR OUTPUT},(COMMA){MANIP JOI STRINGVARCHAR,(NULL)}}}
	{ELSE}
		\\#Take var 'splitter' at the current index and prepend it with a comma and add it to output
		{VAR OUTPUT={MANIP APP {VAR OUTPUT},(COMMA){VAR {VAR INDEX}_splitter}}}
{STATEMENT END}

{STATEMENT NAME FORLOOP_LOGIC}
	\\#numerically check these values
	{NUMERIC}

	\\#If the current index of 'splitter' is less than the length of the 'splitter' array
	{OP1 {VAR INDEX}}
	{CMP lt}
	{OP2 {MANIP SUB {MANIP CNT splitter,(NULL)},1}}
		
		\\#go back to the top of forloop
		{FORLOOP}
	{ELSE}

		\\#do nothing
		(NULL)
{STATEMENT END}
