{FUNCTION NAME FORLOOP}
	\\#Check the nonNull statement
	{VAR TMP={VAR {VAR INDEX}_splitter}}
	{IF ({VAR TMP} -ne )}
		{IF ({VAR TMP} -eq string)}
			\\#Break the current word into a char array, save it as a STRINGVARCHAR array
			{VAR STRINGVAR={VAR {VAR INDEX}_splitter}}
			{MANIP TCA STRINGVAR,STRINGVARCHAR}
	
			\\#Reverse the char array
			{MANIP REV STRINGVARCHAR}
			
			\\#Join with nothing and prepend with a comma
			{VAR OUTPUT={MANIP APP {VAR OUTPUT},(COMMA){MANIP JOI STRINGVARCHAR,(NULL)}}}
		{ELSE}
			{VAR OUTPUT={MANIP APP {VAR OUTPUT},(COMMA){VAR {VAR INDEX}_splitter}}}
		{FI}
	{FI}

	\\#add one to index
	{VAR INDEX++}

	\\#go to FORLOOP_Logic statement
	{VAR TMP_COUNT={EVAL {MANIP CNT splitter,(NULL)}-1}}
	{IF ({VAR INDEX} -lt {VAR TMP_COUNT})}
		{FORLOOP}
	{FI}
{FUNCTION END}
